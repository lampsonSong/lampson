## 任务 2：LLM 错误分类（base.py + agent.py）

**文件**：`src/core/adapters/base.py`、`src/core/agent.py`

**问题**：`base.py` 的 `chat()` 把所有异常都转成 `RuntimeError`，原始错误信息丢失。`agent.py` 的 `_chat_with_fallback` 无法区分哪些错误值得 fallback。

**修复方式**：

### base.py — 使用自定义异常替代 RuntimeError

```python
class LLMError(Exception):
    """LLM 调用失败的基类。"""
    def __init__(self, message: str, original_error: Exception | None = None):
        super().__init__(message)
        self.original_error = original_error

class LLMRetryableError(LLMError):
    """可重试的错误（超时、网络断开、服务端 5xx）。"""
    pass

class LLMRateLimitError(LLMError):
    """频率限制 / 余额不足——可以 fallback 到其他模型重试。"""
    pass

class LLMFatalError(LLMError):
    """不可重试的错误（认证失败、参数错误）。"""
    pass

class LLMContextTooLongError(LLMError):
    """prompt 超长——需要触发压缩而不是 fallback。"""
    pass
```

### base.py — chat() 方法分类抛异常

```python
def chat(self, messages, tools=None):
    kwargs = {"model": self.llm.model, "messages": messages}
    if tools and self.supports_native_tools:
        kwargs["tools"] = tools
        kwargs["tool_choice"] = "auto"
    try:
        return self.llm.client.chat.completions.create(**kwargs)
    except APITimeoutError as e:
        raise LLMRetryableError("LLM 请求超时", original_error=e)
    except APIConnectionError as e:
        raise LLMRetryableError(f"无法连接到 LLM API：{e}", original_error=e)
    except RateLimitError as e:
        raise LLMRateLimitError("API 频率限制或余额不足", original_error=e)
    except httpx.HTTPStatusError as e:
        status_code = e.response.status_code
        if status_code == 401:
            raise LLMFatalError(f"认证失败：{e}", original_error=e)
        elif status_code == 400:
            # 检查是否是 prompt 超长
            body = str(e.response.text) if hasattr(e.response, 'text') else ''
            if 'context' in body.lower() or 'token' in body.lower() or 'length' in body.lower():
                raise LLMContextTooLongError(f"Prompt 超长：{e}", original_error=e)
            raise LLMFatalError(f"请求参数错误：{e}", original_error=e)
        elif 500 <= status_code < 600:
            raise LLMRetryableError(f"服务端错误 {status_code}：{e}", original_error=e)
        else:
            raise LLMFatalError(f"HTTP {status_code}：{e}", original_error=e)
    except httpx.HTTPError as e:
        raise LLMRetryableError(f"网络错误：{e}", original_error=e)
```

### agent.py — _chat_with_fallback 按类型处理

```python
def _chat_with_fallback(self, tools=None):
    self.check_interrupt()
    
    # 1. 先试主模型
    try:
        return self.adapter.chat(self.llm.messages, tools=tools)
    except LLMContextTooLongError:
        # prompt 超长 → 不 fallback，直接上抛让上层处理（触发压缩）
        raise
    except LLMFatalError:
        # 不可重试 → 尝试 fallback
        if not self.fallback_models:
            raise
        logger.warning(f"主模型 {self.llm.model} 调用失败（不可重试），尝试 fallback")
    except (LLMRetryableError, LLMRateLimitError):
        # 可重试 → 尝试 fallback
        if not self.fallback_models:
            raise
        logger.warning(f"主模型 {self.llm.model} 调用失败（可重试），尝试 fallback")
    
    # 2. 依次试 fallback
    for fb_llm, fb_adapter in self.fallback_models:
        try:
            fb_llm.messages = list(self.llm.messages)
            return fb_adapter.chat(fb_llm.messages, tools=tools)
        except LLMContextTooLongError:
            raise  # prompt 超长换模型也没用
        except LLMFatalError:
            continue  # 换下一个
        except (LLMRetryableError, LLMRateLimitError):
            continue  # 换下一个
    
    raise LLMFatalError("所有模型（含 fallback）均调用失败")
```

### agent.py — _run_tool_loop 处理 LLMContextTooLongError

在 `_run_tool_loop` 中，如果收到 `LLMContextTooLongError`，立即触发压缩：
```python
try:
    response = self._chat_with_fallback(tools=self._tools)
except AgentInterrupted:
    raise
except LLMContextTooLongError as e:
    logger.warning(f"Prompt 超长: {e}")
    return "[上下文过长，请使用 /compaction 手动压缩后重试]"
except (LLMError) as e:
    return f"[LLM 错误] {e}"
```
